Q2)

var books = [
  {title:"Inferno",
   author:"Dan Brown",
   alreadyRead: false
  },
  {title:"Sherlock Holmes",
  author:"Conan Doyle",
  alreadyRead: true
  }];

for (var i = 0; i < books.length; i++) {
  var book = books[i];
  var bookstatus = book.title + " by" + book.author;
  if (book.alreadyRead) {
    console.log("You already read " + bookstatus);
  } else {
    console.log("You still need to read " + bookstatus);
  }
}


OUTPUT
You still need to read Inferno by Dan Brown
You already read Sherlock Holmes by Conan Doyle
